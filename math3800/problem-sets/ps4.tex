\begin{homeworkProblem}
	\begin{proof}
		Suppose, for contradiction, that there are only finitely many primes congruent to $3 \pmod{4}$.
		List them as $p_1, p_2, \dots, p_n$ and set
		\[
			P = p_1p_2 \cdots p_n
		\]

		Now define
		\[
			N = 4P - 1.
		\]
		Since we know that 4P is divisible by 4, we can write
		\[
			4P \equiv 0 \pmod{4}
		\]
		We also know that.
		\[
			-1 \equiv 3\pmod{4}
		\]
		So by Theorem 1.3 we can write

		\[
			N \equiv -1 \equiv 3 \pmod{4}.
		\]

		We claim that none of the primes $p_i$ divide $N$. Indeed, for each $i$,
		\[
			N = 4P - 1 \equiv -1 \pmod{p_i},
		\]
		because $p_i \mid P$. Thus $p_i \nmid N$.

		Let
		\[
			N = q_1^{e_1} q_2^{e_2} \cdots q_t^{e_t}
		\]
		be the prime factorization of $N$. Each $q_j$ is an odd prime, so $q_j \equiv 1 \pmod{4}$ or $q_j \equiv 3 \pmod{4}$.

		By Theorem~2.37, the product of numbers each congruent to $1 \pmod{4}$ is itself congruent to $1 \pmod{4}$.
		Since $N \equiv 3 \pmod{4}$, not all $q_j$ can be $\equiv 1 \pmod{4}$. Therefore, at least one $q_j \equiv 3 \pmod{4}$.

		This prime $q_j$ is distinct from the listed primes $p_1, \dots, p_n$, contradicting the assumption that we had found all such primes.

		Hence there must be infinitely many primes congruent to $3 \pmod{4}$.
	\end{proof}

\end{homeworkProblem}

\begin{homeworkProblem}

	\paragraph{(i)}
	\begin{proof}

		Assume, for the sake of contradiction, that $n$ is composite. Then we can write
		\[
			n = ab
		\]
		for integers $a, b > 1$.

		Now apply the factorization identity
		\[
			x^{ab} - 1 = (x^a - 1)\big(x^{a(b-1)} + x^{a(b-2)} + \cdots + x^a + 1\big).
		\]
		Taking $x = 2$, we obtain
		\[
			2^n - 1 = 2^{ab} - 1 = (2^a - 1)\big(2^{a(b-1)} + 2^{a(b-2)} + \cdots + 2^a + 1\big).
		\]

		Because both factors are greater than $1$, this means $2^n - 1$ has a non 1 factorization. But this contradicts our assumption that $2^n - 1$ is prime.

		Therefore $n$ cannot be composite, and so $n$ must be prime.
	\end{proof}

	\paragraph{(ii)}
	\begin{proof}
		Assume for contradiction that $n$ is an odd composite number, and has an odd factor. Then we can write
		\[
			n = ab
		\]
		with $a > 1$ odd and $b \geq 1$.

		Now use the factorization identity provided for when a is odd,
		\[
			x^{ab} + 1 = (x^b + 1)\big(x^{b(a-1)} - x^{b(a-2)} + \cdots - x^b + 1\big).
		\]
		Taking $x = 2$, we obtain
		\[
			2^n + 1 = 2^{ab} + 1 = (2^b + 1)\big(2^{b(a-1)} - 2^{b(a-2)} + \cdots - 2^b + 1\big).
		\]

		We see again that factors are strictly greater than $1$, so $2^n + 1$ has a non 1 factor. But this contradicts our assumption that $2^n + 1$ is prime.

		Therefore $n$ cannot have an odd factor. The only natural numbers without odd factors are the powers of $2$.
	\end{proof}

\end{homeworkProblem}

\begin{homeworkProblem}
	\begin{proof}
		Suppose $n$, $n+2$, and $n+4$ are all prime numbers.
		And we know every integer n is congruent to 0,1, or 2 mod 3

		Based on this lets consider $n$ modulo $3$.

		\begin{itemize}
			\item If $n \equiv 0 \pmod{3}$, then $n$ is divisible by $3$. Since $n$ is prime, the only possibility is $n=3$.

			\item If $n \equiv 1 \pmod{3}$, then $n+2 \equiv 0 \pmod{3}$. Hence $n+2$ is divisible by $3$.
			      If $n+2$ is prime, then $n+2=3 \implies n=1$, which is not prime. Contradiction.

			\item If $n \equiv 2 \pmod{3}$, then $n+4 \equiv 0 \pmod{3}$. Hence $n+4$ is divisible by $3$.
			      If $n+4$ is prime, then $n+4=3 \implies n=-1$, which is not prime. Contradiction.
		\end{itemize}

		Thus, the only possible case is $n=3$. Therefore the only prime triple of the form
		$(n, n+2, n+4)$ is $(3, 5, 7)$.
	\end{proof}


\end{homeworkProblem}

\begin{homeworkProblem}
		We want to solve the system of congruences
		\[
			3x \equiv 2 \pmod{4}, \quad 4x \equiv 1 \pmod{5}.
		\]

We first reduce each congruence to the form \( x \equiv a \pmod{m} \).

\[
\begin{aligned}
x = 0 &\Rightarrow 3(0) = 0 \equiv 0 \pmod{4} \\
x = 1 &\Rightarrow 3(1) = 3 \equiv 3 \pmod{4} \\
x = 2 &\Rightarrow 3(2) = 6 \equiv 2 \pmod{4} 
\end{aligned}
\]

Thus,
\[
x \equiv 2 \pmod{4}
\]

And for the second congruence:

\[
\begin{aligned}
x = 0 &\Rightarrow 4(0) = 0 \equiv 0 \pmod{5} \\
x = 1 &\Rightarrow 4(1) = 4 \equiv 4 \pmod{5} \\
x = 2 &\Rightarrow 4(2) = 8 \equiv 3 \pmod{5} \\
x = 3 &\Rightarrow 4(3) = 12 \equiv 2 \pmod{5} \\
x = 4 &\Rightarrow 4(4) = 16 \equiv 1 \pmod{5} 
\end{aligned}
\]

Thus,
\[
x \equiv 4 \pmod{5}
\]

Now Let:
\[
a = 2, \quad m = 4, \quad b = 4, \quad n = 5
\]

Since \( \gcd(4, 5) = 1 \), the CRT applies, and a unique solution exists modulo \( 20 \).

We use the Extended Euclidean Algorithm to solve:
\[
4u + 5v = 1
\]
Using back-substitution:
\[
1 = 5 - 1 \cdot 4 = (-1)\cdot 4 + 1\cdot 5 \Rightarrow u = -1,\; v = 1
\]

We then compute:
\[
x_0 = a - m \cdot u \cdot (a - b) = 2 - 4 \cdot (-1) \cdot (-2) = 2 - 8 = -6
\]

So our final solution is in the form:
\[
x \equiv -6 \pmod{20} 
\]

\end{homeworkProblem}
